package academy.pocu.comp2500.lab3;

import java.util.ArrayList;

public class ListItem {
    private String text;
    private ArrayList<ListItem> sublistItems = new ArrayList<>();
    private char bulletStyle;
    private ArrayList<Integer> allDepth = new ArrayList<>();
    private ArrayList<ListItem> allList = new ArrayList<>();

    public ListItem(String text) {
        this(text, '*');
    }

    public ListItem(String text, char bulletStyle) {
        this.text = text;
        this.bulletStyle = bulletStyle;
    }

    public String getText() {
        return text;
    }

    public void setText(String str) {
        this.text = str;
    }

    public char getBulletStyle() {
        return bulletStyle;
    }

    public void setBulletStyle(char bulletStyle) {
        this.bulletStyle = bulletStyle;
    }

    public void addSublistItem(final ListItem item) {
        sublistItems.add(item);
    }

    public ListItem getSublistItem(final int index) {
        return sublistItems.get(index);
    }

    public void removeSublistItem(final int index) {
        sublistItems.remove(index);
    }

    private void getRecursive(ListItem item, int depth) {
        allList.add(item);
        allDepth.add(depth);

        if (item.sublistItems != null) {
            ++depth;
            for (ListItem sublist : item.sublistItems) {
                getRecursive(sublist, depth);
            }
        }
    }

    public String toString() {
        StringBuilder sb = new StringBuilder();
        allList.clear();
        allDepth.clear();
        getRecursive(this, 0);
        for (int i = 0; i < allList.size(); ++i) {
            String blank = "    ".repeat(allDepth.get(i));
            sb.append(String.format("%s%c %s%s",
                    blank,
                    allList.get(i).getBulletStyle(),
                    allList.get(i).text,
                    System.lineSeparator()));
        }
        return sb.toString();
    }
}
