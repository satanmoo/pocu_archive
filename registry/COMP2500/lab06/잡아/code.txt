package academy.pocu.comp2500.lab6;

import java.util.ArrayList;

public class Menu {

    protected int price;
    protected boolean isPizza;

    protected int meatCount;
    protected int veggieCount;
    protected int cheeseCount;
    protected int maxMeatCount;
    protected int maxVeggieCount;
    protected int maxCheeseCount;

    protected ArrayList<Dessert> desserts = new ArrayList<>();
    protected ArrayList<Appetizer> appetizers = new ArrayList<>();
    protected ArrayList<MainCourse> mainCourses = new ArrayList<>();

    protected Menu() {
    }

    public int getPrice() {
        return price;
    }

    public boolean isValid() {
        boolean result = true;
        if (isPizza) {
            if (maxMeatCount != Integer.MAX_VALUE) {
                result = result && (this.meatCount == maxMeatCount);
            }
            if (maxVeggieCount != Integer.MAX_VALUE) {
                result = result && (this.veggieCount == maxVeggieCount);
            }
            if (maxCheeseCount != Integer.MAX_VALUE) {
                result = result && (this.cheeseCount == maxCheeseCount);
            }
        } else {
            if (price == 20) {
                result = result && this.desserts.size() == 4;
            }

            if (price == 15) {
                result = result && this.appetizers.size() == 2 && this.desserts.size() > 0;
            }

            if (price == 25) {
                result = result && this.appetizers.size() > 0 && this.mainCourses.size() > 0
                        && this.desserts.size() > 0;
            }
        }
        return result;
    }
}

package academy.pocu.comp2500.lab6;

import java.util.ArrayList;

public class Combo extends Menu {

    protected Combo() {
        super();
    }

    public ArrayList<Appetizer> getAppetizers() {
        assert (appetizers.size() != 0) : "call isValid() first!";
        return appetizers;
    }

    public ArrayList<MainCourse> getMainCourses() {
        assert (mainCourses.size() != 0) : "call isValid() first!";
        return mainCourses;
    }

    public ArrayList<Dessert> getDesserts() {
        assert (desserts.size() != 0) : "call isValid() first!";
        return desserts;
    }
}
