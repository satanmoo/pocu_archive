# 코드보기: C#의 ref 키워드 흉내내기

```asm
    .PROCESSOR 6502
    .ORG $8000

; mul() args
p00=$00
p01=$01
; mmm() related
p10=$02
p11=$03
v10=$04
pmin=$05 ; out
pmax=$07 ; out
min=$1000
max=$1001
    ldx #$FF
    txs

    lda #3
    sta p10

    lda #5
    sta p11

    lda #<min
    sta pmin
    lda #>min
    sta pmin+1

    lda #<max
    sta pmax
    lda #>max
    sta pmax+1

    jsr mmm

end:
    jmp end

mmm: ;(p10,p11,pmin,pmax)->A,(pmin),(pmax)|<X,Y,P>
;=================================================
; summary:   Calculates the product, the minimum
;            and the maximum of p10 and p11
;            
; arguments: p10      The 1st number
;            p11      The 2nd number
;            pmin     The address to store the
;                     minimum value
;            pmax     The address to store the
;                     maximum value
;
; returns:   A        p10 * p11
;            (pmin)   min(p10, p11)
;            (pmax)   max(p10, p11)
;
; modifies:  X
;            Y
;            P
;=================================================
    .SUBROUTINE
    ; mul
    lda p10
    sta p00
    lda p11
    sta p01
    jsr mul

    ; temp store
    sta v10

    ldy #0

    ; min/max
    lda p10
    cmp p11
    bcs .geq
    ; p10 < p11
    sta (pmin),y
    lda p11
    sta (pmax),y
    jmp .ret

.geq: ; p10 >= p11
    sta (pmax),y
    lda p11
    sta (pmin),y

.ret:
    lda v10
    rts
;=================================================

mul: ;(p10,p11) -> A | <X,SR> 
;=================================================
; summary:   Calculates the product of p10 and p11
;            
; arguments: p10  The 1st number
;            p11  The 2nd number
;
; returns:   A    p10 * p11
;
; modifies:  SR
;            X
;=================================================
    .SUBROUTINE
    lda #0
    ldx p01

    clc

.loop:
    beq .ret

    adc p00
    dex
    jmp .loop
.ret:
    rts
;=================================================

    .ORG $FFFC,0
    .WORD $8000
    .WORD $0000
```

> [!abstract] C#의 ref 키워드를 6502 어셈블리로 흉내내기
>
> **C#에서의 ref**
> ```csharp
> void Mul(ref byte p00, ref byte p01) // 포인터로 원본 수정
> ```
>
> **6502에서의 구현**
> ```asm
> ; mul() args
> p00=$00    ; 제로페이지에 포인터 배치
> p01=$01
> ```
>
> 6502에는 ref 키워드가 없지만,
> 제로페이지에 **값이 저장된 주소(포인터)**를 전달하면
> 함수 내부에서 `(zp),y`로 원본을 직접 수정할 수 있다.
>
> 이 기법은 **반환값이 여러 개**일 때 특히 유용하다.
> A 레지스터 하나로는 1바이트밖에 반환할 수 없지만,
> 포인터를 여러 개 넘기면 호출자의 메모리를 **직접 수정**하여
> 사실상 여러 값을 반환하는 효과를 낸다.

 > [!example] mmm() — 레지스터 반환 vs 포인터(ref) 반환 비교
>
> **함수 시그니처**
> ```
> mmm: ;(p10,p11,pmin,pmax)->A,(pmin),(pmax)|<X,Y,P>
> ```
>
> | 반환 방식 | 이전 버전 | 이번 버전 |
> |----------|----------|----------|
> | 곱 (product) | A 레지스터 | A 레지스터 (동일) |
> | 최솟값 (min) | X 레지스터 | **(pmin)** 포인터로 반환 |
> | 최댓값 (max) | Y 레지스터 | **(pmax)** 포인터로 반환 |
>
> [[010. zp를 이용한 인자 전달]]에서 본 이전 버전은 A, X, Y **3개 레지스터를 모두 소진**하여 반환했다.
> 이번 버전은 pmin, pmax 포인터를 통해 호출자의 메모리에 직접 써넣으므로
> **X, Y 레지스터를 반환용으로 쓰지 않아도 된다.**
>
> `<X,Y,P>`가 꺾쇠 안에 있는 이유는
> 함수 내부에서 X, Y, P가 **변경(clobber)되지만 반환값이 아니라는 의미**다.

> [!important] min/max 비교 및 포인터를 통한 반환
>
> **cmp의 동작: A - M**
> ```asm
> lda p10        ; A = p10
> cmp p11        ; A - M = p10 - p11 (결과 버림, 플래그만 갱신)
> bcs .geq       ; C=1이면 .geq로 점프
> ```
>
> | 플래그 | 의미 |
> |--------|------|
> | C=1 (carry set) | 받아내림 없음 → p10 ≥ p11 |
> | C=0 (carry clear) | 받아내림 발생 → p10 < p11 |
>
> `bcs` = **B**ranch if **C**arry **S**et
> 뺄셈에서 캐리가 세트 = 받아내림이 없다 = A ≥ M이므로
> p10 ≥ p11일 때 `.geq`로 점프한다.
>
> **p10 < p11인 경우 (bcs 분기 안 함)**
> ```asm
> ; p10 < p11
> sta (pmin),y    ; *pmin = p10 (A에 이미 p10이 있음)
> lda p11
> sta (pmax),y    ; *pmax = p11
> jmp .ret
> ```
>
> **p10 ≥ p11인 경우 (.geq)**
> ```asm
> ; p10 >= p11 → 반대로 저장
> sta (pmax),y    ; *pmax = p10
> lda p11
> sta (pmin),y    ; *pmin = p11
> ```
>
> `sta (pmin),y` / `sta (pmax),y`는 포인터가 가리키는 주소에 직접 써넣으므로
> 호출자의 min, max 변수가 **함수 내부에서 변경**된다 (ref 반환).

> [!note] ldy #0으로 (zp),y를 사실상 절대 간접 참조로 사용
> ```asm
> ldy #0
> sta (pmin),y    ; *(pmin + 0) = *(pmin)
> ```
>
> Y=0이면 오프셋이 없으므로 `(zp),y`는 사실상 `(zp)` 처럼 동작한다.
> 즉 포인터가 가리키는 주소 **그 자체**에 접근하는 절대 간접 참조와 동일하다.
>
> 6502에 `lda (zp)` / `sta (zp)` 같은 순수 간접 모드는 없으므로,
> **Y=0을 세팅하고 (zp),y를 쓰는 것이 유일한 우회 방법**이다.
