## 트리는 그래프의 일종이다

- DAG
    - 방향 비순환 그래프

## 그래프 용어

- 차수:
    - 노드에 연결된 변의 개수

- 루프:
    - 자기 자신으로 돌아오는 간선을 가진 노드

## 방향 그래프 vs 무방향 그래프

- 무방향 그래프는 꼬리-머리, 머리-꼬리 양방향 이동 가능

- 일부 간선만 방향을 가지고 있어도, 방향 그래프로 간주
    - 모든 간선이 방향을 가질 필요 없음

## 순환 그래프 vs 비순환 그래프

- 어떤 노드를 떠나서 다시 그 노드로 돌아오는 경로가 1개 이상 존재하면 순환 그래프

## 방향 그래프의 인접 행렬 특징

- 대각선 대칭을 보장하지 못함

## 인접 행렬의 장점과 단점

- 장점:
    - 탐색 속도
        - from a to b 변의 존재(가중치)를 O(1)에 검색
- 단점:
    - 변의 수와 관계없이 노드 수에 비례한 공간 복잡도
        - O(N^2)
    - 인접 행렬 생성할 때 시간복잡도 O(N^2)

## 인접 리스트의 장점과 단점

- 장점:
    - 인접 행렬에 비해 작은 공간 복잡도
        - 변의 수에 맞게 공간 활용
        - O(N + E)
    - 연결 리스트의 삽입/삭제 시간복잡도 적용
        - 장점인가?

- 단점:
    - from a to b 변의 존재(가중치)를 찾을 때 행을 순회해야해서 O(1)의 시간복잡도가 아님

## 인접 행렬 구현하기

```java
import java.util.ArrayList;
import java.util.List;

public class Program {

    private int[][] getAdjacentMatrix(List<Node> nodes) {
        int[][] adjacentMatrix = new int[nodes.size()][nodes.size()];

        for (Node node : nodes) {
            for (Node neighbor : node.neighbors) {
                adjacentMatrix[node.index][neighbor.index] = 1;
            }
        }

        return adjacentMatrix;
    }

    private static class Node {
        final int index;
        final char value;
        final List<Node> neighbors = new ArrayList<>();

        Node(final int index, final char value) {
            this.index = index;
            this.value = value;
        }
    }
}
```

## 방향 그래프 DFS로 모든 노드 방문하게 구현

- 노드에서 방문 여부를 기억하는 방식으로 구현

```java
import java.util.ArrayList;
import java.util.List;

public class Program {

    public static void main(String[] args) {
        Node A = new Node('A');
        Node B = new Node('B');
        Node C = new Node('C');
        Node D = new Node('D');
        Node E = new Node('E');
        Node F = new Node('F');

        A.neighbors.add(E);
        A.neighbors.add(C);

        C.neighbors.add(E);
        C.neighbors.add(D);

        E.neighbors.add(F);

        B.neighbors.add(F);

        searchDepthFirst(List.of(A, B, C, D, E, F));
    }

    private static void searchDepthFirst(final List<Node> nodes) {
        for (Node node : nodes) {
            if (!node.visited) {
                node.visited = true;
                searchRecursive(node);
            }
        }
    }

    private static void searchRecursive(final Node current) {
        System.out.println(current.value);

        for (Node neighbor : current.neighbors) {
            if (!neighbor.visited) {
                neighbor.visited = true;
                searchRecursive(neighbor);
            }
        }
    }

    private static class Node {
        final char value;
        final List<Node> neighbors = new ArrayList<>();
        boolean visited = false;

        Node(final char value) {
            this.value = value;
        }
    }
}
```

```java
        Node D = new Node('D');
        Node E = new Node('E');
        Node F = new Node('F');

        A.neighbors.add(E);
        A.neighbors.add(C);

        C.neighbors.add(A);
        C.neighbors.add(E);
        C.neighbors.add(D);

        E.neighbors.add(A);
        E.neighbors.add(C);
        E.neighbors.add(F);

        F.neighbors.add(E);
        F.neighbors.add(B);

        B.neighbors.add(F);

        D.neighbors.add(C);

        searchDepthFirst(List.of(A, B, C, D, E, F));
    }

    private static void searchDepthFirst(final List<Node> nodes) {
        final HashSet<Node> discovered = new HashSet<>();

        for (Node node : nodes) {
            if (!discovered.contains(node)) {
                discovered.add(node);
                searchRecursive(node, discovered);
            }
        }
    }

    private static void searchRecursive(final Node current, final HashSet<Node> discovered) {
        for (Node neighbor : current.neighbors) {
            if (!discovered.contains(neighbor)) {
                discovered.add(neighbor);
                searchRecursive(neighbor, discovered);
            }
        }
        System.out.println(current.value);
    }

    private static class Node {
        final char value;
        final List<Node> neighbors = new ArrayList<>();

        Node(final char value) {
            this.value = value;
        }
    }
}
```

```java
package academy.pocu.comp3500.lab10.app;

import java.util.ArrayDeque;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Deque;
import java.util.HashMap;
import java.util.List;

public class Program {

    public static void main(String[] args) {
        List<Course> courses = createCourseGraph();
        List<Course> sorted = sortTopological(courses);
        sorted.forEach(System.out::println);
    }

    private static List<Course> sortTopological(final List<Course> courses) {
        final HashMap<Course, Integer> indegree = new HashMap<>();
        final Deque<Course> dq = new ArrayDeque<>();
        for (final Course course : courses) {
            indegree.putIfAbsent(course, 0);
            for (final Course next : course.nextCourses) {
                indegree.put(next, indegree.getOrDefault(next, 0) + 1);
            }
        }

        for (var e : indegree.entrySet()) {
            if (e.getValue() == 0) {
                dq.addLast(e.getKey());
            }
        }

        List<Course> result = new ArrayList<>();
        while (!dq.isEmpty()) {
            Course course = dq.removeLast();
            for (final Course next : course.nextCourses) {
                indegree.put(next, indegree.get(next) - 1);
                if (indegree.get(next) == 0) {
                    dq.addLast(next);
                }
            }
            result.add(course);
        }

        return result;
    }
```

## 송수관 깔기 문제

- SCC로 그래프 압축
- SCC들을 모두 연결하는 최소 수의 변을 구하기

## SCC 수학적 정의

- C에 속한 모든 정점 쌍 u,v에 대해서 u -> v, v -> u인 가장 큰 집합

## 코사라주 알고리즘 구현

```java
import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Set;
import java.util.Stack;

public class Program {

    public static void main(String[] args) {
        final List<SCC> compressed = findSccKosaraju(createNode());
    }

    private static List<SCC> findSccKosaraju(final List<Node> nodes) {
        List<Node> dfsResult = searchDepthFirst(nodes);
        getTransposeGraph(nodes);

        return compressGraph(dfsResult);
    }

    private static List<SCC> compressGraph(final List<Node> dfsResult) {
        final List<SCC> compressed = new ArrayList<>();
        final HashSet<Node> discovered = new HashSet<>();
        final HashMap<Node, SCC> nodeSCCHashMap = new HashMap<>();
        for (final Node node : dfsResult) {
            if (!discovered.contains(node)) {
                final SCC scc = new SCC();
                discovered.add(node);
                searchDepthFirstReverse(node, discovered, scc);
                compressed.add(scc);
                scc.nodeSet.forEach(n -> nodeSCCHashMap.put(n, scc));
            }
        }
        connectCompressedGraph(compressed, nodeSCCHashMap);
        return compressed;
    }

    private static void connectCompressedGraph(final List<SCC> compressed, final HashMap<Node, SCC> nodeSCCHashMap) {
        compressed.forEach(scc ->
                scc.nodeSet.forEach(node ->
                        node.edges.forEach(edge -> {
                                    assert edge.from == node;
                                    final SCC scc1 = nodeSCCHashMap.get(edge.from);
                                    final SCC scc2 = nodeSCCHashMap.get(edge.to);
                                    if (scc1 != scc2) {
                                        scc2.entries.add(edge.to);
                                        scc1.successors.add(scc2);
                                    }
                                }
                        )
                )
        );
    }

    private static void searchDepthFirstReverse(final Node current, final HashSet<Node> discovered, final SCC scc) {
        scc.nodeSet.add(current);
        for (final Edge rEdge : current.reversedEdges) {
            if (!discovered.contains(rEdge.to)) {
                discovered.add(rEdge.to);
                searchDepthFirstReverse(rEdge.to, discovered, scc);
            }
        }
    }

    private static void getTransposeGraph(final List<Node> nodes) {
        nodes.forEach(node -> node.edges.forEach(edge -> {
            Edge reversed = new Edge(edge.to, edge.from);
            edge.to.reversedEdges.add(reversed);
        }));
    }

    private static List<Node> searchDepthFirst(final List<Node> nodes) {
        final Stack<Node> stack = new Stack<>();
        final HashSet<Node> discovered = new HashSet<>();

        for (final Node node : nodes) {
            if (!discovered.contains(node)) {
                discovered.add(node);
                searchDepthFirstRecursive(node, discovered, stack);
            }
        }

        final List<Node> result = new ArrayList<>();
        while (!stack.isEmpty()) {
            result.add(stack.pop());
        }

        return result;
    }

    private static void searchDepthFirstRecursive(final Node current, final HashSet<Node> discovered, final Stack<Node> outStack) {
        for (final Edge edge : current.edges) {
            if (!discovered.contains(edge.to)) {
                discovered.add(edge.to);
                searchDepthFirstRecursive(edge.to, discovered, outStack);
            }
        }
        outStack.add(current);
    }

    private static List<Node> createNode() {
        final Node a = new Node(3);
        final Node b = new Node(2);
        final Node c = new Node(0);
        final Node d = new Node(5);
        final Node e = new Node(4);
        final Node f = new Node(1);
        final Node g = new Node(6);

        a.edges.add(new Edge(a, b));

        b.edges.add(new Edge(b, a));

        c.edges.add(new Edge(c, b));
        c.edges.add(new Edge(c, e));

        d.edges.add(new Edge(d, c));

        e.edges.add(new Edge(e, d));

        f.edges.add(new Edge(f, c));

        g.edges.add(new Edge(g, f));

        return List.of(a, b, c, d, e, f, g);
    }

    static final class Node {
        final int data;
        final List<Edge> edges = new ArrayList<>();
        final List<Edge> reversedEdges = new ArrayList<>();

        Node(final int data) {
            this.data = data;
        }

        @Override
        public String toString() {
            return "Node{" +
                    "data=" + data +
                    '}';
        }
    }

    static final class Edge {
        final Node from;
        final Node to;

        Edge(final Node from, final Node to) {
            this.from = from;
            this.to = to;
        }
    }

    static final class SCC {
        final Set<Node> entries = new HashSet<>();
        final Set<Node> nodeSet = new HashSet<>();
        final List<SCC> successors = new ArrayList<>();
    }
}
```
